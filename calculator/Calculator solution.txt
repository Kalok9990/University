import java.util.Scanner;


public class Calculator {

    
	public static void main(String args[]) {

        
		// Create a Scanner object
        
		Scanner reader = new Scanner(System.in);

        

		// Read the operation
        
		System.out.print("Enter your operation: ");
        
		String input = reader.next();

        

		// Parse input string
        
		char operator = input.charAt(2);
        
		String a_string = input.substring(0, 2);
        
		String b_string = input.substring(3, 5);

        

		// Parse strings into integers
        
		int a = Integer.parseInt(a_string);
        
		int b = Integer.parseInt(b_string);

        
		long result;

        
		switch(operator){

            
		case '+':
                
		result = a + b;
                
		System.out.println(a + " + " + b + " = " + result);
                
		break;

            

		case '-':
                
		result = a - b;
                
		System.out.println(a + " - " + b + " = " + result);
                
		break;

            

		case '*':
                
		result = a * b;
                
		System.out.println(a + " * " + b + " = " + result);
                
		break;

            

		case '^':
                
		result = pow(a, b);
                
		System.out.println(a + " ^ " + b + " = " + result);
                
		break;

            

		default:
                
		System.out.println("Unknown operator: " + operator);

        }

    
}

    
public static long pow(int base, int exponent) {


        long result = 1;


        for (int i = 0; i < exponent; i++) {

            result = result * base;  // result *= base;

        }


        return result;
    
}

}
